PROBLEM: Two Sum
// Traverse the array once and maintain map..
#include <iostream>
#include <vector>
using namespace std;

int main() {
vector<int> arr = {1, 2, 3, 4, 5};

int sum = 0;
for (int num : arr) {
sum += num;
}

cout << sum << endl; // Output: 15
return 0;
}
